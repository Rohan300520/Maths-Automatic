Trigonometric Identities

\begin{equation*} \frac{1+tan\left(x\right)}{1-tan\left(x\right)}=\frac{\cos\left(x\right)+\sin\left(x\right)}{\cos\left(x\right)-\sin\left(x\right)} \end{equation*}
\begin{equation*} \cot\left(\frac{t}{2}\right)^2=\frac{(1+\cos\left(t\right))}{(1-\cos(t))} \end{equation*}
\begin{equation*} \text{ verify }\tanθ+\cotθ=\secθ\cscθ \end{equation*}

Mathematical Induction

\begin{equation*} \text{ prove by induction }\sum_{j\mathop{=}1}^nj=\frac{1}{2}n(n+1) \end{equation*}
\begin{equation*} \text{ prove by induction }\sum_{i\mathop{=}0}^n2^i=2^{n+1}-1 \end{equation*}
\begin{equation*} \text{ induction 3 divides }n^3-7n+3 \end{equation*}
\begin{equation*} \text{ show with induction }2n+7 < n+7^2\text{ where }n\ge1 \end{equation*}
\begin{equation*} \text{ prove by induction }\sum_{k=0}^nx^ky^{-k+n} \begin{pmatrix}n\\k\end{pmatrix}=(x+y)^n \end{equation*}
